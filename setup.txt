Create 
/usr/local/bin/ipwatch.sh

sudo chmod +x /usr/local/bin/ipwatch.sh

Create /etc/systemd/system/ipwatch.service

[Unit]
Description=Check eno1 and wg0 IPs at boot and notify via ntfy
After=network.target

[Service]
Type=simple
ExecStart=/usr/local/bin/ipwatch.sh
Restart=no

[Install]
WantedBy=multi-user.target



enable the service: 

sudo systemctl daemon-reload
sudo systemctl enable ipwatch.service
